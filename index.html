架构

<!DOCTYPE webPages version="1.0">
<set>
	<title>Hello, World!</title>
	<meta>charset="utf-8"</meta>
	<script src="https://unpkg.com/vue@2.6.11/dist/vue.js" type="text/javascript" charset="utf-8"></script>
</set>

<template>
	<div id="#app">
		<h1 style="text-align: center" v-text="msg"></h1>
	</div>
</template>

<style>
	h1{
		font-size:36px;
	}
</style>

<script>
	var vm=new Vue({
		el:"#app",
		data(){
			return{
				msg:'Hello, World!'
			}
		},
		methods:{
			
		}
	});
</script>
